부모 클래스 예제 : RockPaperScissors

import java.util.Scanner;  
public class RockPaperScissors { 클래스 시작
    private int player; 변수 선언
    private final int ROCK = 0;
    private final int PAPER = 1;
    private final int SCISSORS = 2; 		상수 선언
    public RockPaperScissors() {
        this.player = SCISSORS;
    } 		생성자
    public void select( Scanner scan ) {
        String input="";
        do {
            System.out.print( "가위, 바위, 보 중 하나를 입력하세요 : ");
            input = scan.next();
        } while( !this.player.equals("가위") && !this.player.equals("바위") && !this.player.equals("보") );
        scan.close();
        if ( input.equals( "가위" ) )
            this.player = SCISSORS;
        else if ( input.equals( "바위" ) )
            this.player = ROCK;
        else
            this.player = PAPER;
    } 		select 메서드1
    public void select() {
        this.player = (int)( Math.random() * 3 );
    } 		select 메서드2
    public String toString() {
        if ( this.player == SCISSORS )
            return "가위";
        else if ( this.player == ROCK )
            return "바위";
        else
            return "보";
    } 		toString 메서드
    public String toInteger() {
        return this.player;
    } 		toInteger 메서드
    public boolean equals( RockPaperScissors counterpart ) {
        return this.player == counterpart.toInteger();
    } 		equals 메서드
    public boolean win( RockPaperScissors counterpart ) {
        return win( counterpart.toInteger() );
    } 		win 메서드1
    public boolean win( int counterpart ) {
        if ( ( ( this.player == SCISSORS ) && ( counterpart == PAPER ) )
          || ( ( this.player == PAPER ) && ( counterpart == ROCK ) )
          || ( ( this.player == ROCK ) && ( counterpart == SCISSORS ) ) )
            return true;
        else
            return false;
    } 		win 메서드2
} 클래스 끝

interface 상속

interface   상수와 추상 메소드로만 구성되며, 이들의 접근권한은 모두 public

implements   부모 인터페이스의 모든 추상 메서드를 상속받아 자식 클래스에서 구현

추상메서드   메서드 이름, 반환자료형, 전달인자만 명시되고 실행 내용은 없는 메서드

public interface RemoteControl { 인터페이스 시작
    public final boolean ON = true;
    public final boolean OFF = false; 		상수
    public String clickPower();
    public String clickUp();
    public String clickDown();
    public String clickLeft();
    public String clickRight(); 		추상 메서드
} 인터페이스 끝
package remoteControl;  
public class TV implements RemoteControl { 자식 클래스 시작
    private boolean power;
    private int channel;
    private int volume; 		변수
    public TV() {
        this.power = OFF;
        this.channel = 100;
        this.volume = 10;
    } 		생성자
    public String clickPower() {
        if( this.power == OFF ) {
            this.power = ON;
            return "전원을 켭니다";
        }
        else {
            this.power = OFF;
            return "전원을 끕니다";
        }
    } 		오버라이딩 메서드
    public String clickUp() {
        return "현재 채널은 " + ( ++this.channel ) + "번입니다";
    } 		오버라이딩 메서드
    public String clickDown() {
        return "현재 채널은 " + ( --this.channel ) + "번입니다";
    } 		오버라이딩 메서드
    public String clickLeft() {
        return "현재 음량은 " + ( --this.volume ) + "입니다";
    } 		오버라이딩 메서드
    public String clickRight() {
        return "현재 음량은 " + ( ++this.volume ) + "입니다";
    } 		오버라이딩 메서드
} 자식 클래스 끝

다형성

다형성   메서드 호출시 참조하는 클래스 객체에 따라 다른 메서드를 호출 가능
부모 클래스/인터페이스의 메서드를 자식 클래스에서 오버라이딩하면, 메서드 호출시 자식 클래스의 메서드 실행
 
RemoteControl remocon = new TV();
remocon.clickUp(); TV 채널 + 1 실행
remocon = new RoboticVacuumCleaner();
remocon.clickUp(); 이동방향 직진 실행
remocon = new AirConditioner();
remocon.clickUp(); 희망온도 +1 실행

import java.util.Scanner;  
public class Main { 클래스 시작
    public static void main(String[] args) {
        Scanner scan = new Scanner( System.in );
        String result = "";
        RemoteControl remocon[] = { new TV(), new AirConditioner(), new RoboticVacuumCleaner() };
        do {
            System.out.print( "리모컨의 상, 하, 좌, 우, 전원, 종료 중 하나를 입력하세요 : ");
            result = scan.next();
            if ( result.equals( "전원" ) ) {
                for ( int i = 0; i < remocon.length; i++ ) {
                    System.out.println( remocon[i].clickPower() );
                }
            }
            else if ( result.equals( "상" ) ) {
                for ( int i = 0; i < remocon.length; i++ ) {
                    System.out.println( remocon[i].clickUp() );
                }
            }
            else if ( result.equals( "하" ) ) {
                for ( int i = 0; i < remocon.length; i++ ) {
                    System.out.println( remocon[i].clickDown() );
                }
            }
            else if ( result.equals( "좌" ) ) {
                for ( int i = 0; i < remocon.length; i++ ) {
                    System.out.println( remocon[i].clickLeft() );
                }
            }
            else if ( result.equals( "우" ) ) {
                for ( int i = 0; i < remocon.length; i++ ) {
                    System.out.println( remocon[i].clickRight() );
                }
            }
        } while( !result.equals( "종료" ) );
    } 		프로그램 시작

abstract 클래스

abstract 클래스   abstract 메서드를 포함하는 클래스
메서드가 모두 구현되지 않아 객체 생성 불가능하고, 메서드를 모두 구현한 자식 클래스의 객체는 생성 가능

abstract 메서드   선언되어 있으나 구현되어 있지 않은 추상 메소드
 
abstract class RemoteControl2 { 클래스 시작

            this.power = OFF;
