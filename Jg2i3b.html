  TV  RemoteControl  
{				 
	 boolean power;	  
	 int channel;  
	 int volume;		  
	 
	 TV()					  
	{ 
		.power = OFF;		  
		.channel = 100;  
		.volume = 10;		  
	} 
	 
	 String clickPower()	  
	{ 
		( .power  OFF )	  
		{ 
			.power = ON;		  
			 "TV 전원을 켭니다"; 
		} 
		  
		{ 
			.power = OFF;		  
			 "TV 전원을 끕니다"; 
		} 
	} 
	 
	 String clickUp()		  
	{ 
		 "현재 채널은 " + ( ++.channel ) + "번입니다";		  
	} 
	 
	 String clickDown()				  
	{ 
		 "현재 채널은 " + ( --.channel ) + "번입니다";		  
	} 
	 
	 String clickLeft()			  
	{ 
		 "현재 음량은 " + ( --.volume ) + "입니다";		  
	} 
	 
	 String clickRight()		  
	{ 
		 "현재 음량은 " + ( ++.volume ) + "입니다";		  
	} 
}
