  TV  RemoteControl  
{				 
	 boolean power;	  
	 int channel; 		  
	 int volume;			  
	 
	 TV() { 
		.power = OFF;			  
		.channel = 100; 	 
		.volume = 10;		 
	} 
	 
	 String clickPower() { 
		( .power  OFF ) { 
			.power = ON;		 				  
			 "TV 전원을 켭니다"; 
		} 
		 { 
			.power = OFF;		 			 	 
			 "TV 전원을 끕니다"; 
		} 
	} 
	 
	 String clickUp() { 
		 "현재 채널은 " + ( ++.channel ) + "번입니다";		 				  
	} 
	 
	 String clickDown() { 
		 "현재 채널은 " + ( --.channel ) + "번입니다";		 			  
	} 
	 
	 String clickLeft() { 
		 "현재 음량은 " + ( --.volume ) + "입니다";		 			  
	} 
	 
	 String clickRight() { 
		 "현재 음량은 " + ( ++.volume ) + "입니다";		 			  
	} 
}

  RemoteControl_All_Main  
{ 
	 static void main (String[] args ) { 
		Scanner scan =  Scanner( System.in ); 
		String result = ""; 
 
		RemoteControl rc[] = { 
		                         TV(), 
		                         AirConditioner(), 
		                         RoboticVacuumCleaner()  
		                     }; 
		 
		 { 
			System.out.print( "리모컨의 상, 하, 좌, 우, 전원, 종료 중 하나를 입력하세요 : "); 
			result = scan.next(); 
 
			 ( result.( "전원" ) ) { 
				 ( int i = 0; i < rc.length; i++ ) { 
					System.out.println( 
					                    rc[i].clickPower() );	  
				} 
			} 
			 ( result.( "상" ) ) { 
				 ( int i = 0; i < rc.length; i++ ) { 
					System.out.println( 
					                    rc[i].clickUp() );	  
				} 
			} 
			 ( result.( "하" ) ) { 
				 ( int i = 0; i < rc.length; i++ ) { 
					System.out.println( 
					                    rc[i].clickDown() );	  
					} 
				} 
			 ( result.( "좌" ) ) { 
				 ( int i = 0; i < rc.length; i++ ) { 
					System.out.println( 
					                    rc[i].clickLeft() );	  
				} 
			} 
			 ( result.( "우" ) ) { 
				 ( int i = 0; i < rc.length; i++ ) { 
					System.out.println(

				}
